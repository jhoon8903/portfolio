  • S보드 위 다수 블록이 각자 경로 탐색 → 상하좌우 순회 검증 시 블록 수만큼 연산 폭증, 프레임 부하

Assets/@Projects/99.Scripts/Systems/PathFinding/Jobs/PathFindJob.cs
[BurstCompile]
public struct PathFindJob : IJob
{
    public int GridWidth;
    public int GridHeight;
    [ReadOnly] public NativeArray<bool> WalkableMap;
    public int2 Start;
    public int2 Goal;
    public NativeList<int2> PathResult;

    public void Execute()
    {
        PathResult.Clear();
        if (!IsValid(Start) || !IsValid(Goal)) return;
        var startIndex = ToIndex(Start);
        var goalIndex  = ToIndex(Goal);
        if (!WalkableMap[startIndex] ||
            !WalkableMap[goalIndex]) return;

        var queue    = new NativeQueue<int2>(Allocator.Temp);
        var visited  = new NativeHashSet<int>(128, Allocator.Temp);
        var cameFrom = new NativeHashMap<int,int>(128, Allocator.Temp);
        queue.Enqueue(Start);
        visited.Add(startIndex);
        while (queue.TryDequeue(out int2 cur))
        {
            if (cur.x == Goal.x && cur.y == Goal.y) break;
            for (var i = 0; i < 4; i++)
            {
                var nb = cur + GridUtility.FourDirections[i];
                if (!IsValid(nb)) continue;
                var ni = ToIndex(nb);
                if (visited.Contains(ni)) continue;
                if (!WalkableMap[ni]) continue;
                visited.Add(ni);
                queue.Enqueue(nb);
                cameFrom[ni] = ToIndex(cur);
            }
        }
        // cameFrom 역추적으로 PathResult 복원 (생략)

        // Allocator.Temp 컨테이너는 끝에서 직접 해제 → 누수 0
        queue.Dispose();
        visited.Dispose();
        cameFrom.Dispose();
    }
}
💡 Burst+Job으로 BFS를 메인스레드 밖에서 돌리고, Allocator.Temp NativeContainer를 Execute 끝에서 직접 Dispose해 GC/누수 0. WalkableMap은 [ReadOnly]로 안전 보장.

public class SlotFactory : ISlotFactory
{
    private readonly IObjectResolver _resolver;
    private readonly IPool _pool;

    public SlotFactory(
        IObjectResolver resolver, IPool pool)
    {
        _resolver = resolver;
        _pool = pool;
    }

    public (SlotModel model, SlotView view) Create(
        string id, int index, Vector3 pos,
        SlotState state, Transform parent = null)
    {
        var model = new SlotModel();
        model.Initialize(id, index, state);
        var view = _pool.Get<SlotView>();   // 풀에서 대여
        if (view == null) return (model, view);
        _resolver.Inject(view);             // 대여 View에 DI 주입
        if (parent != null)
            view.transform.SetParent(parent);
        view.transform.position = pos;
        view.Model = model;
        return (model, view);
    }

    public void Release(SlotModel model, SlotView view)
    {
        if (view != null)
        {
            view.Model = null;
            _pool.Return(view);             // 풀로 반납
        }
        model?.Reset();
    }
}
💡 Model은 new로, View는 IPool에서 대여(Get)하고 IObjectResolver로 즉시 DI 주입 — Instantiate 비용 제거. Release에서 Model 참조를 끊고 Return해 더블 릴리즈/누수를 방지.

public class CommandController : IDisposable
{
    private readonly Queue<ICommand> _queue = new();
    private readonly object _lock = new();
    private bool _isProcessing;
    private MUpdateLoop _loop;

    public CommandController()
    {
        _loop = ModuleContainer.Get<MUpdateLoop>();
        if (_loop != null) _loop.OnTick += ProcessNext;
    }

    public void Enqueue(ICommand command)
    {
        if (command == null) return;
        lock (_lock) { _queue.Enqueue(command); }
    }
    // 매 프레임 MUpdateLoop.OnTick에서 호출
    private void ProcessNext()
    {
        if (_isProcessing || _queue.Count == 0) return;
        ICommand command = null;
        lock (_lock)
        {
            if (_queue.Count > 0)
                command = _queue.Dequeue();
        }
        if (command == null) return;
        try
        {
            _isProcessing = true;
            command.Execute();
        }
        finally
        {
            command.Dispose();      // 커맨드 수명 종료
            _isProcessing = false;
        }
    }
}
💡 턴/상태 변경을 ICommand로 캡슐화해 큐에 쌓고, MUpdateLoop.OnTick에서 한 프레임 한 개씩 순차 실행 — _isProcessing 가드로 재진입을 막고 lock으로 enqueue/dequeue를 보호, finally에서 Dispose해 커맨드 수명을 닫음.

Main Project — Wiggle Puzzle ⑤ · WebGL Porting

모바일 게임을 "설치 없이 브라우저에서" — WebGL 포팅

앱스토어 밖(일반 H5 · 토스 앱인토스 · 카카오)에서 도는 빌드로. 게임 로직(퍼즐·BFS)은 그대로 두고, 깨지는 건 게임에 붙은 주변 부품과 빌드 설정 — "재작성"이 아니라 "교체·정리" 프로젝트.

"세 개의 다른 벽" — 통과 지점도, 에러도 다르다

1 · 에디터 컴파일

스크립트가 에디터에서 컴파일되는가. (콘솔 CS 에러)

2 · 플레이어 빌드

WebGL IL2CPP→wasm. 에디터는 통과해도 여기서 깨짐 — 예: ICMP Ping.

3 · 런타임

브라우저에서 실제로 도는가. 빌드 성공해도 깨짐 — 예: 공격적 스트리핑.

무엇이 깨졌나 — 빌드 로그로 단계적 규명(추측 X)

  • WebGL 정의 심볼 미설정 — 사본의 define이 비어 프레임워크 전체가 #if 뒤로 사라짐(첫 증상은 엉뚱한 CS0246)
  • 모바일 네이티브 SDK(광고·분석·결제·소셜) — WebGL 미지원 + 스토어 리젝 리스크
  • 네이티브 API — UnityEngine.Ping(ICMP) 등은 WebGL 플레이어 어셈블리에 없음
  • URP 버전 정합성 — GlobalSettings "not at last version"으로 빌드 전처리기가 거부
  • jslib 누락 — 햅틱 [DllImport] 호출의 JS 구현이 없어 wasm 링크 undefined symbol

각 벽은 통과 단계가 달라, 한 번에 하나씩 바꾸고 매번 실제 빌드·실행으로 검증.

MVC · MVVM 패턴 — 같은 골격, 다른 살

게임 로직(엔티티)은 MVC, UI 화면은 MVVM으로 통일한다. 두 패턴 모두 베이스 클래스가 옵저버·바인딩·라이프사이클을 미리 처리해, 게임은 살(Refresh/Bind)만 채운다.

MVC — 엔티티 / 게임오브젝트

  • Model : Observable — 값 변경 시 Raise()로 옵저버 통지
  • View : BaseView<TModel> — Model 바인딩·풀 라이프사이클 자동, RefreshView()만 구현
  • Factory/Controller — 생성·반납을 한 데스크가 소유(생성기가 찍음)

MVVM — UI 화면

  • ViewModel : ViewModelBase — [Inject]로 서비스 주입, Initialize/CleanUp
  • View : UI<TViewModel> — ViewModel을 DI 주입받아 BindViewModel()에서 구독
  • 등록 — 씬 스코프가 ViewModel 자동등록 + View 주입(배선 0줄)

// === MVC : Model (Observable) ===
public abstract class Observable : IObservable
{
    private List<IObserver> _observers;
    protected void Raise()                 // 값 변경 후 호출 → 뷰 갱신
    {
        if (_observers == null) return;
        for (int i = 0; i < _observers.Count; i++)
            _observers[i]?.OnChanged(this);
    }
    public void AddObserver(IObserver o)    => (_observers ??= new()).Add(o);
    public void RemoveObserver(IObserver o) => /* null-슬롯 처리 */ ;
}

// 게임 모델은 이렇게만 :
public sealed class CoinModel : Observable
{
    private int _amount;
    public int Amount => _amount;
    public void SetAmount(int v) { _amount = v; Raise(); }
}

// === MVC : View (BaseView) — 바인딩·풀 라이프사이클 자동 ===
public abstract class BaseView<TModel> : PoolableView, IObserver
    where TModel : class, IObservable
{
    public TModel Model { get; set; }       // setter가 Bind/Unbind 구동
    public void OnChanged(IObservable o)    // 모델 변경 콜백
    { if (o == Model) RefreshView(); }
    public override void OnInactive()       // 풀 반납 시 자동 언바인딩
    { Model = null; base.OnInactive(); }

    protected abstract void RefreshView();  // ← 게임은 이것만 구현
}
// === MVVM : ViewModel (ViewModelBase) ===
public abstract class ViewModelBase : IViewModel
{
    public abstract void Initialize();      // 데이터 로드·구독
    public abstract void CleanUp();         // 해제

    // [Inject] public void Construct(ISaveSystem s) { ... }
}

// === MVVM : View (UI<TViewModel>) ===
public abstract class UI<TViewModel> : UIView
    where TViewModel : IViewModel, new()
{
    [Inject] public void Construct(TViewModel vm) => ViewModel = vm;
    protected TViewModel ViewModel { get; private set; }

    public override bool Initialize()
    {
        ViewModel.Initialize();
        BindViewModel();                    // ← 게임은 여기서 구독
        return true;
    }
    protected abstract void BindViewModel();
}

// 화면은 이렇게만 :
public sealed class ShopView : UI<ShopViewModel>
{
    protected override void BindViewModel()
    {
        ViewModel.OnGoldChanged += Refresh;
        Refresh();
    }
}

// 화면 스코프 : ViewModel 자동등록 + 모든 UIView 주입 + 풀 정리
protected override void Configure(IContainerBuilder builder)
{
    // 프로젝트의 모든 ViewModel 한 번에 등록
    ViewModelInstaller.Install(
        builder, GetType().Assembly, typeof(ViewModelBase));

    builder.RegisterBuildCallback(container =>
    {
        _nav  = container.Resolve<IUINavigationService>();
        _pool = container.Resolve<IPool>();

        // 씬의 모든 UIView 를 루트 순회로 주입
        // → 화면을 추가해도 배선 코드 0줄
        foreach (var root in
                 gameObject.scene.GetRootGameObjects())
        foreach (var v in root
                 .GetComponentsInChildren<UIView>(true))
            container.InjectGameObject(v.gameObject);
    });
}

            foreach (var h in root.GetComponentsInChildren
                     <AssetReceiveHolder>(true))
                container.InjectGameObject(h.gameObject);

            foreach (var m in root.GetComponentsInChildren
                     <MonitorBase>(true))
                container.InjectGameObject(m.gameObject);
        }
    });
}
💡 화면·엔티티·모니터가 같은 한 패턴(루트 순회 InjectGameObject)으로 자동 배선된다 — 생성기는 이 데스크를 가리키기만 하면 되고, 항목을 추가해도 등록 코드가 늘지 않는다.

AI Workflow ① — Thesis

설계는 사람, 구현은 AI

AI 도구는 모두에게 같습니다 — Claude Code·Codex·Gemini를 같은 날 받았죠. 변별은 도구가 아니라 무엇을 먹이고(규칙·명세) 무엇을 검증·소유하느냐, 그리고 AI에 끌려가지 않느냐에서 납니다. AI는 매끄럽게 틀리고(환각), 정보가 많을수록 사람은 판단을 AI에 넘기기 쉽기에(자동화 편향) — 저는 끌려가지 않도록 규칙을 먼저 만듭니다.

# 결정 순간, 정보를 쏟지 말고 결정을
# 사용자에게 되돌려준다.
# 목적: 과부하로 AI에 끌려가는 것
#       (자동화 편향·앵커링) 방지

## 결정 순간에 항상 적용
1. 번역 우선 — 전문정보 나열 말 것.
   각 선택지에서 무엇이 깨지는지 예시로.
   이해 못 하면 설명이 덜 된 것.

2. 결정을 사용자 영역으로 — "어느 기술이
   옳은가"(AI)가 아니라 "내 목표·제약에
   뭐가 맞나"(사용자)로. 질문 2~3개 제시.
3. 자기반박 동봉 — 반대 선택지의 가장
   강한 논리 + "내 추천이 틀리려면?"

4. 되돌리기 비용으로 신중함 보정 — 쉬운
   결정은 "그냥 고르고 진행", 비가역
   (세이브·데이터포맷·아키텍처)만 깊이.

5. 사실은 실행으로 검증 — 말로 설득 말고
   최소 실행·공식문서로. 환각 가능성 땐
   단정 말고 불확실성 표시.

6. "끌려감" 감지 시 멈춤 — 이해가 아니라
   압도되어 따라오면 속도 늦춰 다시 설명.

## 톤  전문성 과시 말 것.
#  AI=무엇이 가능한가, 사용자=무엇이 맞나

## Critical Rules  (10)
 1. 불필요한 메서드 만들지 말 것 — 쓰는 것만

## Core Responsibilities  (매 루프)

AI Workflow ③ — The Human Lane

AI가 못 하는 칸 — 눈 · 검증 · 소유

👁 연출은 눈으로 판단

코어 로직은 AI가 짜도 — 타격 이펙트 0.15초냐 0.25초냐, 점수 팝업이 파티클보다 먼저냐 나중이냐, 이징이 ease-out이냐 back-out이냐. 명세에도 없고 테스트로도 안 잡혀 빌드 돌려 눈·손맛으로 정합니다. AI는 제 빌드를 못 봅니다.

✓ 검증 · 소유

AI 출력은 그대로 안 믿고, 세이브처럼 깨지면 데이터가 날아가는 곳은 테스트로 정조준. 라인 단위로 설명 못 하는 코드는 출시 안 합니다.

끌려가지 않기도 사람의 칸 — 그럴듯한 답일수록 실행·테스트로 확인하고(환각), AI의 첫 답에 갇히지 않게 반대안을 먼저 받습니다(앵커링·자동화 편향).
